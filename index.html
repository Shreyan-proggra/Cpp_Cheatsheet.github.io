Switch case syntax

    switch (expression)
    {

}

Make a simple calculator in C++

    
#include<iostream>
using namespace std;

int main() {
    int a, b;
    char op;
    cout<<"Enter the value you want to calculate"<< endl;
    cin>>a>>op>>b;
    switch (op)
    {
    case '+':
        cout<< a+b;
        break;
    case '-':
        cout<< a-b;
        break;
    case '*':
        cout<< a*b;
        break;
    case '/':
        cout<< a/b;
        break;
    
    default:
        break;
    }
    return 0;
}
